107 RequestBuilder &RequestBuilder::updateFromRelativeUrl(const QUrl &url, bool mergeQuery, bool keepFragment)
108 {
109  auto cUrl = buildUrl();
110  d->base = cUrl.resolved(url);
111  if(d->base.host() != cUrl.host()) {
112  qWarning() << "URL host changed from"
113  << cUrl.host()
114  << "to"
115  << d->base.host();
116  }
117  //clear all the rest
118  d->version = QVersionNumber();
119  d->user.clear();
120  d->pass.clear();
121  d->path.clear();
122  if(mergeQuery) {
123  QUrlQuery query(url.query());
124  foreach(auto item, query.queryItems())
125  d->query.addQueryItem(item.first, item.second);
126  } else
127  d->query = QUrlQuery(url.query());
128  if(!keepFragment)
129  d->fragment.clear();
130  return *this;
131 }

216 QUrl RequestBuilder::buildUrl() const
217 {
218  auto url = d->base;
219 
220  auto pathList = url.path().split(QLatin1Char('/'), QString::SkipEmptyParts);
221  if(!d->version.isNull())
222  pathList.append(QLatin1Char('v') + d->version.normalized().toString());
223  pathList.append(d->path);
224  url.setPath(QLatin1Char('/') + pathList.join(QLatin1Char('/')) + (d->trailingSlash ? QStringLiteral("/") : QString()));
225 
226  if(!d->user.isNull())
227  url.setUserName(d->user);
228  if(!d->pass.isNull())
229  url.setPassword(d->pass);
230  if(!d->query.isEmpty())
231  url.setQuery(d->query);
232  if(!d->fragment.isNull())
233  url.setFragment(d->fragment);
234 
235  return url;
236 }
